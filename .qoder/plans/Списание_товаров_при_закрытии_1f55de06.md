# Изменение логики списания товаров при закрытии брони

## Проблема
Сейчас при каждом обновлении бронирования товары сначала возвращаются на склад, потом снова списываются. Нужно:
- При создании/изменении брони: только резервировать (проверять наличие, но не списывать)
- При закрытии брони (статус "закрыт"): списывать товары со склада и создавать документ реализации

## Задача 4: Добавить эндпоинты для документов реализации

**Файл:** `backend/app/routers/documents_realization.py` (создать новый)

```python
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session, joinedload
from typing import List
from app.database import get_db
from app.models import RealizationDocument
from app.schemas import RealizationDocumentRead

router = APIRouter(prefix="/admin/documents/realization", tags=["Documents - Realization"])

@router.get("/", response_model=List[RealizationDocumentRead])
def get_realization_documents(db: Session = Depends(get_db)):
    return db.query(RealizationDocument)\
             .options(
                 joinedload(RealizationDocument.items)
                 .joinedload(RealizationDocumentItem.product)
             )\
             .all()

@router.get("/{doc_id}", response_model=RealizationDocumentRead)
def get_realization_document(doc_id: int, db: Session = Depends(get_db)):
    doc = db.query(RealizationDocument)\
             .options(
                 joinedload(RealizationDocument.items)
                 .joinedload(RealizationDocumentItem.product)
             )\
             .filter(RealizationDocument.id == doc_id)\
             .first()
    if not doc:
        raise HTTPException(status_code=404, detail="Document not found")
    return doc

@router.delete("/{doc_id}", status_code=204)
def delete_realization_document(doc_id: int, db: Session = Depends(get_db)):
    doc = db.query(RealizationDocument).filter(RealizationDocument.id == doc_id).first()
    if not doc:
        raise HTTPException(status_code=404, detail="Document not found")
    
    # Возвращаем товары на склад
    for item in doc.items:
        product = db.query(models.Product).filter(models.Product.id == item.product_id).first()
        if product:
            product.total_quantity += item.quantity
    
    # Удаляем бронь
    db.query(models.Reservation).filter(
        models.Reservation.reservation_id == doc.reservation_id
    ).delete()
    
    db.delete(doc)
    db.commit()
    return
```

## Задача 5: Создать SQL скрипт для новых таблиц

**Файл:** `backend/add_realization_documents.sql`

```sql
CREATE TABLE realization_documents (
    id SERIAL PRIMARY KEY,
    date DATE NOT NULL DEFAULT CURRENT_DATE,
    reservation_id INTEGER NOT NULL REFERENCES reservations(reservation_id),
    client_name VARCHAR(100) NOT NULL,
    client_phone VARCHAR(20) NOT NULL,
    total_amount FLOAT NOT NULL DEFAULT 0.0
);

CREATE TABLE realization_document_items (
    id SERIAL PRIMARY KEY,
    document_id INTEGER NOT NULL REFERENCES realization_documents(id) ON DELETE CASCADE,
    product_id INTEGER NOT NULL REFERENCES products(id),
    quantity INTEGER NOT NULL,
    price FLOAT NOT NULL
);

CREATE INDEX idx_realization_doc_reservation ON realization_documents(reservation_id);
CREATE INDEX idx_realization_doc_items_doc ON realization_document_items(document_id);
```
